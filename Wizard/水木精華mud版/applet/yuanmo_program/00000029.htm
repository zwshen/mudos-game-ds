//math.cc 
 
#include"mathtbl.h" 
 
CLookupTable::CLookupTable() 
{ 
        CreateLookupTable(); 
} 
CLookupTable::~CLookupTable() 
{ 
} 
void CLookupTable::CreateLookupTable() 
{ 
        for(int i=0;i<numAngles;i++){ 
                Sin12[i]=(long)(4096.0*sin(i*2*M_PI/numAngles)+0.5); 
                Cos12[i]=(long)(4096.0*cos(i*2*M_PI/numAngles)+0.5); 
                Sin[i]=Sin12[i]>>2; 
                Cos[i]=Cos12[i]>>2; 
        } 
}
