                  ^ Z 
                  | 
                  3--------2 
                  |  tri1 / | 
                  |     /   | 
                  |   /     | 
                  |  / tri2 |  X 
                  1--------4--> 
*/ 
class ObjectChain 
{ 
public: 
        Unit *unit; 
        ObjectChain *next; 
public: 
        ObjectChain() 
                {unit=NULL;next=NULL;} 
        ~ObjectChain() 
                {Clear();} 
        void Add(Unit *newunit) 
                {if(next!=NULL)next->Add(newunit); 
                 else{ 
                        unit=newunit; 
                        next=new ObjectChain;if(next==NULL)exit(3); 
                 } 
                } 
        void Clear() 
                {if(unit==NULL)return; 
                 next->Clear(); 
                 delete next; 
                 unit=NULL; 
                 next=NULL; 
                } 
};
