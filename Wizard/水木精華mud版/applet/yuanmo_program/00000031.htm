#include"mission.h" 
int Objective::ProcessResult() 
{ 
        int result=PROCESS_ONTHEWAY; 
        switch(accomplishcondition) 
        { 
                case CONDITION_NONE: 
                break; 
                case CONDITION_ALIVE: 
                        if(unit->m_healthCondition<HC_DYING){ 
                                result=PROCESS_SUCCEED; 
                        } 
                break; 
                case CONDITION_DESTROYED: 
                        if(unit->m_healthCondition>=HC_DYING){ 
                                result=PROCESS_SUCCEED; 
                        } 
                break; 
                case CONDITION_REACH: 
                        long x=unit->m_position.x; 
                        long y=unit->m_position.z; 
                        long minx=MIN(dwords[0],dwords[2]); 
                        long maxx=MAX(dwords[0],dwords[2]); 
                        long miny=MIN(dwords[1],dwords[3]); 
                        long maxy=MAX(dwords[1],dwords[3]); 
                        if(x>=minx&&x<=maxx&&y>=miny&&y<=maxy){ 
                                result=PROCESS_SUCCEED; 
                        } 
                break; 
        } 
        switch(failcondition) 
        { 
                case CONDITION_NONE: 
                break; 
                case CONDITION_ALIVE: 
                        if(unit->m_healthCondition<HC_DYING){ 
                                result=PROCESS_FAIL; 
                        } 
                break; 
                case CONDITION_DESTROYED: 
                        if(unit->m_healthCondition==HC_DEAD){ 
                                result=PROCESS_FAIL; 
                        } 
                break; 
                case CONDITION_REACH: 
                        long x=unit->m_position.x; 
                        long y=unit->m_position.z; 
                        long minx=MIN(dwords[0],dwords[2]); 
                        long maxx=MAX(dwords[0],dwords[2]); 
                        long miny=MIN(dwords[1],dwords[3]); 
                        long maxy=MAX(dwords[1],dwords[3]); 
                        if(x>=minx&&x<=maxx&&y>=miny&&y<=maxy){ 
                                result=PROCESS_FAIL; 
                        } 
                break; 
        } 
        return result; 
} 
 
CGoal::CGoal() 
{ 
        m_numObjects=0; 
        m_numfinished=0; 
        m_numTotalObjectives=0; 
} 
CGoal::~CGoal() 
{ 
} 
int CGoal::AddObjective(Objective &obj) 
{ 
        if(m_numObjects==maxobjects){ 
                return false; 
        } 
        m_objList[m_numObjects]=obj; 
        m_numObjects++; 
        if(obj.accomplishcondition!=CONDITION_NONE) 
                m_numTotalObjectives++; 
        return m_numObjects-1; 
} 
void CGoal::DeleteObjective(ID num) 
{ 
        m_numObjects--; 
        for(int i=num;i<m_numObjects;i++){ 
                m_objList[i]=m_objList[i+1]; 
        } 
} 
int CGoal::ProcessTrace() 
{ 
        Objective *obj; 
        for(int i=0;i<m_numObjects;i++){ 
                int result=m_objList[i].ProcessResult(); 
                if(result==PROCESS_SUCCEED){ 
                        m_numfinished++; 
                        DeleteObjective(i); 
                        if(m_numfinished==m_numTotalObjectives){ 
                                return PROCESS_SUCCEED; 
                        } 
                } 
                else if(result==PROCESS_FAIL){ 
                        DeleteObjective(i); 
                        return PROCESS_FAIL; 
                } 
        } 
        return PROCESS_ONTHEWAY; 
}
