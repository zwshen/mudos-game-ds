你的npc和標準npc不一樣哦,看看把幾句carry_object()刪了，或把 
setup()挪到carry_object()之前看看 
【 在 zixia (紫霞仙子) 的大作中提到: 】 
︰ 轅馬如下： 
︰ #include <ansi.h>        
︰ inherit NPC; 
︰ inherit F_DEALER; 
︰ void create() 
︰ {        
︰         set_name( HIG "小茜" NOR , ({"xiao xi", "shinv"}) ); 
︰         set("id","xiaoxi"); 
︰         set("nickname", YEL "丫環" NOR); 
︰         set("gender","女性"); 
︰         set("age",16); 
︰         set("str",50); 
︰         set_skill("literate",150); 
︰         set("no_get_from", 1); 
︰         carry_object("/open/zixia/armor/bangle")->wear(); 
︰         setup(); 
︰ } 
︰ void init()        
︰ {               
︰         object master, me, xiaoxi; 
︰         master = find_player("zixia"); 
︰         xiaoxi = this_object(); 
︰         me = this_player(); 
︰         ::init(); 
︰ }        
︰ }
