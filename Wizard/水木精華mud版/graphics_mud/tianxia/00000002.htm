private string _get_waction_desc( string *sActionTypes ) 
{ 
        string sResult, sActionType; 
 
        sActionType = choose_one(sActionTypes); 
 
        if( random(2) )  
                sResult = "$N" + choose_one( ({ "揮動", "舞動", "揮舞", "舉起" }) ) + "$w，"; 
        else 
                sResult = "$N" + choose_one( ({ "將手中的", "將", "用", "的" }) ) + "$w"; 
 
        if( random(2) )  
                sResult += choose_one( ({ "住", "向", "朝", "對準" }) ) + "$n的$l" 
                        + choose_one( ({ "直" + sActionType, sActionType + "將" }) ) + "過去。"; 
        else if( random(2) )  
                sResult += choose_one( ({ "住", "向", "朝", "對準" }) ) + "$n的$l" 
                        + choose_one( ({ "用力一", "狠狠一", "一陣亂"  }) ) + sActionType + "。"; 
        else  
                sResult += sActionType + "向$n的$l。"; 
 
        return sResult; 
} 
 
private class Action _get_weapon_action( string sActionName ) 
{ 
        class Action cAction = new( class Action ); 
        string sAction; 
 
        switch(sActionName) 
        { 
                case "抹" : 
                        cAction->sAction = _get_waction_desc( ({ "削", "劃", "抹" }) ); 
                        cAction->sDamageType = "割傷"; 
                        break;                 
 
                case "斬" : 
                        cAction->sAction = _get_waction_desc( ({ "砍", "斬", "削" }) ); 
                        cAction->sDamageType = "砍傷"; 
                        cAction->iDamage = 5; 
                        cAction->iDodge = 5; 
                        cAction->iDefense = -5; 
                        break;                 
 
                case "砍" : 
                        cAction->sAction = _get_waction_desc( ({ "劈", "砍" }) ); 
                        cAction->sDamageType = "劈傷"; 
                        cAction->iDamage = 10; 
                        cAction->iDodge = 10; 
                        cAction->iDefense = -10; 
                        break;                 
 
                case "劈" : 
                        cAction->sAction = _get_waction_desc( ({ "劈", "砍" }) ); 
                        cAction->sDamageType = "劈傷"; 
                        cAction->iDamage = 20; 
                        cAction->iDodge = 20; 
                        cAction->iDefense = -20; 
                        break;                 
 
                case "刺" : 
                        cAction->sAction = _get_waction_desc( ({ "刺" }) ); 
                        cAction->sDamageType = "刺傷"; 
                        break;                 
 
                case "扎" : 
                        cAction->sAction = _get_waction_desc( ({ "刺",  "戳", "扎" }) ); 
                        cAction->sDamageType = "扎傷"; 
                        cAction->iDamage = 10; 
                        cAction->iDodge = 10; 
                        cAction->iDefense = -10; 
                        break;                 
 
                case "捅" : 
                        cAction->sAction = _get_waction_desc( ({ "刺", "扎", "捅" }) ); 
                        cAction->sDamageType = "扎傷"; 
                        cAction->iDamage = 20; 
                        cAction->iDodge = 20; 
                        cAction->iDefense = -20; 
                        break;                 
 
                case "打" : 
                        cAction->sAction = _get_waction_desc( ({ "打", "擊" }) ); 
                        cAction->sDamageType = "砸傷"; 
                        break; 
 
                case "擊" : 
                        cAction->sAction = _get_waction_desc( ({ "擊", "砸" }) ); 
                        cAction->sDamageType = "砸傷"; 
                        cAction->iDamage = 10; 
                        cAction->iDodge = 10; 
                        cAction->iDefense = -10; 
                        break;                 
 
                case "砸" : 
                        cAction->sAction = _get_waction_desc( ({ "擊", "砸" }) ); 
                        cAction->sDamageType = "砸傷"; 
                        cAction->iDamage = 20; 
                        cAction->iDodge = 20; 
                        cAction->iDefense = -20; 
                        break;                 
 
                case "錘擊" : 
                        cAction->sAction = _get_waction_desc( ({ "擊", "砸" }) ); 
                        cAction->sDamageType = "挫傷"; 
                        cAction->iDamage = 10; 
                        cAction->iDodge = 10; 
                        cAction->iDefense = -10;

                case "錘砸" :

                        cAction->sDamageType = "鉤傷";

                        cAction->iDodge = 20; 
                        cAction->iDefense = -20;

                case "輕叉" :

                case "叉" : 
                        cAction->sAction = _get_waction_desc( ({ "叉" }) ); 
                        cAction->sDamageType = "叉傷"; 
                        cAction->iDamage = 10; 
                        cAction->iDodge = 10; 
                        cAction->iDefense = -10; 
                        break;                 
 
                case "重叉" : 
                        cAction->sAction = _get_waction_desc( ({ "叉" }) ); 
                        cAction->sDamageType = "叉傷"; 
                        cAction->iDamage = 20; 
                        cAction->iDodge = 20; 
                        cAction->iDefense = -20; 
                        break;                 
 
                case "抽" : 
                        cAction->sAction = _get_waction_desc( ({ "抽", "擊" }) ); 
                        cAction->sDamageType = "抽傷"; 
                        cAction->iDamage = 0; 
                        cAction->iDodge = -20; 
                        cAction->iDefense = 30; 
                        break; 
 
                case "擲" : 
                        cAction->sAction = "$N將$w對準$n的$l射了過去。"; 
                        cAction->sDamageType = "刺傷"; 
                        break; 
 
                case "射" : 
                        cAction->sAction = "$N彎弓搭箭，只聽“嗖”的一聲，$w已射到$n的$l。";

                default : 
                        cAction->sAction = _get_waction_desc( ({ "打", "擊" }) ); 
                        cAction->sDamageType = "擊傷"; 
                        break; 
        } 
         
        return cAction; 
}
